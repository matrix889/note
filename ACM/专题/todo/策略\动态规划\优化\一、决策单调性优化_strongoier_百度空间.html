策略\动态规划\优化\一、决策单调性优化

各种优化的实现参考汪一宁的《1D/1D动态规划优化初步》。


f(x) = min{f(i) + w(i, x)}(0 <= i < x)


对于任意x' > x，令f(x')的最优决策为k'，f(x)的最优决策为k，若k' < k，有

f(x') = f(k') + w(k', x') <= f(k) + w(k, x')

f(k') - f(k) <= w(k, x') - w(k', x')

f(x) = f(k) + w(k, x) <= f(k') + w(k', x)

f(k') - f(k) >= w(k, x) - w(k', x)

w(k, x) - w(k', x) <= w(k, x') - w(k', x')

w(k, x) + w(k', x') <= w(k, x') + w(k', x)

与四边形不等式矛盾。


因 此，当w满足四边形不等式 w(i, j) + w(i + 1, j + 1) <=& nbsp;w(i, j + 1) + w(i + 1, j) (i <= j)时，有决策单调不减。

而要证w满足四边形不等式，只要证w(i, j + 1) - w(i, j)单调不增即可。


【例1】BZOJ 1010 玩具装箱


题目大意：


给 出标准L(1 <= L <= 10^7)和N(1 <= N& nbsp;<= 50000)个玩具，长度分别为c_i(1 <= C_i <=& nbsp;10^7)，现在要依次把它们装入箱中，从第i个到第j个装在同一箱中的代价为 (j - i + sigma{c_k}(i <= k <=& nbsp;j) - L)^2，求最小总代价。


算法分析：


令 f(x)表示到第x个为止的最小总代价，则f(x) = min{f(i) + w(i +& nbsp;1, x)}(0 <= i < x)，其中w(i, j)& nbsp;= (j - i + sigma{c_k}(i <= k& nbsp;<= j) - L)^2。

易证w满足四边形不等式，故可以进行决策单调性优化，时间复杂度O(NlogN)。

#include <cstdio>
   
typedef long long LL;
   
const int MAXN = 55555;
   
int t, c[MAXN], q[MAXN], p[MAXN];
LL s[MAXN], f[MAXN];
   
inline LL sqr(LL x) {
    return x * x;
}
   
inline LL calc(int i, int j) {
    return f[i] + sqr(j - i - 1 + s[j] - s[i] - t);
}
   
int main() {
    int n, l = 1, r = 1;
    scanf("%d%d", &n, &t);
    for (int i = 1; i <= n; ++i) {
        scanf("%d", &c[i]);
        s[i] = s[i - 1] + c[i];
    }
    p[1] = n;
    for (int i = 1; i <= n; ++i) {
        for (; i > p[l]; ++l);
        f[p[l - 1] = i] = calc(q[l], i);
        for (; l <= r && calc(i, p[r - 1] + 1) < calc(q[r], p[r - 1] + 1); --r);
        if (l <= r) {
            int lo = p[r - 1] + 1, hi = p[r] + 1;
            while (lo + 1 < hi) {
                int mid = lo + hi >> 1;
                if (calc(i, mid) < calc(q[r], mid))
                    hi = mid;
                else
                    lo = mid;
            }
            p[r] = lo;
        }
        if (p[r] < n)
            q[++r] = i, p[r] = n;
    }
    printf("%lld\n", f[n]);
    return 0;
}


【例2】BZOJ 1563 诗人小G


题目大意：


有 N(1 <= N <= 100000)个诗句需要被依次排版为若干行，一行内可以有若干个诗句， 相邻诗句之间有一个空格。定义行标准长度L(1 <= L <= 3000000)，每行的不协调 度为|实际长度 - L|^P(1 <= P <= 10)，求最小不协调度 之和。


算法分析：


令 c_i表示第i个诗句的长度，f(x)表示到第x个为止的最小总代价，则f(x) = min{f(i) +  w(i + 1, x)}(0 <= i < x)，其中 w(i, j) = |j - i + sigma{c_k}(i& nbsp;<= k <= j) - L|^P。

对w满足四边形不等式的证明参见http://www.byvoid.com/blog/noi-2009-poet，故可以进行决策单调性优化，时间复杂度O(NlogN)。

需要注意的是，题目要求答案在10^18以内才输出，但在运算过程中是可能超出long long范围的，因此我们必须使用long double进行中间过程的运算，而不能直接把10^18作为运算过程的上界。

#include <cstdio>
#include <cstdlib>
#include <cstring>
  
typedef long double LD;
  
const LD INF = 1e18;
const int MAXN = 111111;
  
char str[33];
LD f[MAXN];
int t, v, a[MAXN], s[MAXN], p[MAXN], q[MAXN];
  
inline LD pow(int x) {
    LD ret = 1;
    for (int i = 1; i <= v; ++i, ret *= x);
    return ret;
}
  
inline LD calc(int i, int j) {
    return f[i] + pow(abs(j - i - 1 + s[j] - s[i] - t));
}
  
inline void work() {
    int n, l = 1, r = 1;
    scanf("%d%d%d", &n, &t, &v);
    for (int i = 1; i <= n; ++i) {
        scanf("%s", str);
        s[i] = s[i - 1] + (a[i] = strlen(str));
    }
    p[0] = q[1] = 0, p[1] = n;
    for (int i = 1; i <= n; ++i) {
        for (; i > p[l]; ++l);
        f[p[l - 1] = i] = calc(q[l], i);
        for (; l <= r && calc(i, p[r - 1] + 1) < calc(q[r], p[r - 1] + 1); --r);
        if (l <= r) {
            int lo = p[r - 1] + 1, hi = p[r] + 1;
            while (lo + 1 < hi) {
                int mid = lo + hi >> 1;
                if (calc(i, mid) < calc(q[r], mid))
                    hi = mid;
                else
                    lo = mid;
            }
            p[r] = lo;
        }
        if (p[r] < n)
            q[++r] = i, p[r] = n;
    }
    if (f[n] <= INF)
        printf("%lld\n", (long long)f[n]);
    else
        puts("Too hard to arrange");
    puts("--------------------");
}
  
int main() {
    int t;
    scanf("%d", &t);
    while (t--)
        work();
    return 0;
}
