单调队列+DP斜率优化

      最近一段时间学习了一下单调队列在DP中的简单应用，以前只是知道一些简单的DP，后来发现，如果数据范围变大后，直接DP是不能比较理想地解决问题的。 开始学的时候觉得挺难的，后来一点点从简单到复杂就不那么麻烦了。

开始可以去看看网上有关单调队列斜率优化的论文。

http://www.docin.com/p-47950655.html 基本看懂了就可以开始做题目了。

http://acm.fzu.edu.cn/problem.php?pid=1894这道题很好地诠释了单调队列，可以用来检测你对单调队列是否真得理解了，可以算是入门题，不过最好自己建立队列，这样对于以后的学习更方便。

 # include <iostream>
 const int n_max=1000005 ;
 int rp[n_max] ;
 int q[n_max] , head , tail ;
 int main()
 {
    int T , i , j , first;
    char t[10] , name[10] ;
    scanf("%d" , &T) ;
    while(T-- && scanf("%s" , t))
    {
       head=0 , tail=-1 , i=0 , first=1 ;
       while(scanf("%s" , t))
       {
          if(t[0]=='E')break ;
          else if(t[0]=='C')//建立单调递减队列
          {
             scanf("%s %d",name , &rp[++i]) ;
             while(head<=tail && rp[q[tail]]<rp[i])tail-- ;
             q[++tail]= i ;
          }
          else if(t[0]=='Q')//输出队首元素（值最大的元素）
          {
             if(head>tail)printf("-1\n") ;
             else printf("%d\n",rp[q[head]]) ;
          }
          else if(t[0]=='G')//这里的删除不是删除单调队列的首元素
          {
             if(first<=i)
             {
               if(first==q[head])head++ ;
               first++ ;
             }
          }
       }
    }
  return 0 ;
 }

 

http://poj.org/problem?id=2823这是另外一道单调队列的题目，也很好，在单纯的单调队列的基础上又增加了一些条件。

 

# include <iostream>
using namespace std;
const int n_max=1000005 ;
int a[n_max] ;
int q[n_max] , head , tail ;
int main()
{
  int n , len ;
  int i , j ;
  scanf("%d %d" , &n ,&len ) ;
  for(i=1 ;i<=n ; i++) scanf("%d",&a[i]) ;
  head=1 , tail=0 ;
  for(i=1 ;i<=n ;i++)
  {
     while(head<=tail && a[q[tail]]>a[i])  tail-- ;//单调递增
     q[++tail]= i;
     while(head<=tail && q[head]<=i-len)head++ ;//要保证长度
     if(i>=len)printf("%d ", a[q[head]]) ;       
  }
  printf("\n") ;
 
  head=1 , tail=0 ;
  for(i=1 ;i<=n ;i++)
  {
     while(head<=tail && a[q[tail]]<a[i])tail-- ;//单调递减
     q[++tail]=i ;
     while(head<=tail && q[head]<=i-len) head++ ;
     if(i>=len) printf("%d ",a[q[head]]);       
  }
  printf("\n") ;
  
  return  0 ;   
}

 http://acm.hdu.edu.cn/showproblem.php?pid=3415一样还是单调队列的简单应用，不过很奇怪，杭电上明明说数据n<100000;可是我定义成100010时居然RE了。（小笨蛋，如果是环的话，数组长度肯定翻倍啊，呵呵呵呵呵。。。。。）

# include <iostream>
using namespace std ;
const int n_max=200010 ;
int a[n_max] , sum[n_max] ;
int q[n_max] , head , tail ;
int main()
{
  int T , n , len , m;
  int i , ans , start , end ;
  scanf("%d", &T) ;
  while(T--)
  {
     scanf("%d %d",&n,&len) ;
     sum[0]=0 ;
     for(i=1 ; i<=n ;i++)
     {
        scanf("%d", &a[i]) ;
        sum[i]=sum[i-1]+a[i] ;       
     }
     for(i=n+1 ; i<n+len ;i++) sum[i]=sum[i-1]+a[i-n] ;
     m=n+len-1 ;
     head=1 , tail=0 ; ans=-0x3fffffff ;
     for(i=1 ;i<=m ; i++)
     {
        while(head<=tail && sum[q[tail]]>sum[i-1])tail-- ;//建一个单调递增序列
        q[++tail]=i-1 ;
        while(head<=tail && q[head]<i-len)head++ ;
        if(sum[i]-sum[q[head]]>ans)
        {
           ans=sum[i]-sum[q[head]];
           start=q[head]+1 ;
           end=i ;                          
        }
     }
     if(end>n)end%=n ;
     printf("%d %d %d\n",ans , start , end) ;
  }
 
  return  0 ;   
}

http://acm.hdu.edu.cn/showproblem.php?pid=3401这是一道单调队列优化的DP，其实每天的状态只有3种：(状态是第i天有j只股的最大值)

1 不买也不买，那么dp[i][j]=dp[i-1][j] ,

2 买入一些股票 dp[i][j]=max(dp[i][j] , dp[i-w-1][k]-(j-k)*b[i])

3 卖出一些股票 dp[i][j]=max(dp[i][j] , dp[i-w-1][k]+(k-j)*s[i])

但是需要优化，因为如果原来有j只股的最大值已经大于当前的操作了，那么这些就没必要做比较了。

# include <iostream>
using namespace std ;
const int inf= 0x6fffffff ;
const int n_max=2005 ;
int b[n_max] , s[n_max] ,bnum[n_max] , snum[n_max] ;
int f[n_max][n_max] ;
struct node
{
  int num , mon ;      
}q[n_max];
int ncase , n , m , w ;

int max(int x , int y)
{
   if(x<y)return y ;
   return x ;   
}
int min(int x ,int y)
{
   if(x>y)return y ;
   return x ;   
}
int main()
{
  scanf("%d",&ncase) ;
  int i , j , k ;
  while(ncase--)
  {
     scanf("%d %d %d",&n ,&m,&w) ;
     for(i=1 ;i<=n ;i++)  scanf("%d %d %d %d",&b[i] , &s[i] , &bnum[i] , &snum[i]) ;
    
     for(i=0 ; i<=n ;i++)
        for(j=0 ; j<=m ;j++)  f[i][j]=-inf ;
       
     for(i=1 ; i<=n ;i++)//初始化
     {
        k=min(m , bnum[i]) ;
        f[i][0]=0 ;       
        for(j=1 ; j<=k ; j++) f[i][j]= -j*b[i] ;
     }
    
     for(i=2 ; i<=n ;i++)
     {
        for(j=0 ; j<=m ;j++)   f[i][j]=max(f[i][j] , f[i-1][j]) ;//不卖不买
       
        if(i<w+2)  continue ;
       
        int l=0 , r=0 ;
        int pre=i-w-1;
        for(j=0 ; j<=m ; j++)//买
        {
           while(l<r && q[r-1].mon-(j-q[r-1].num)*b[i]<f[pre][j])r--;
           q[r].num=j ;
           q[r++].mon=f[pre][j] ;
           while(l<r && q[l].num+bnum[i]<j) l++ ;    
           if(l<r) f[i][j]=max(f[i][j] , q[l].mon-(j-q[l].num)*b[i]) ;
        }
        l=r=0 ;
        for(j=m ;j>=0 ;j--)//卖
        {
           while(l<r && q[r-1].mon+(q[r-1].num-j)*s[i]<f[pre][j]) r-- ;
           q[r].num= j ;
           q[r++].mon=f[pre][j] ;       
           while(l<r && q[l].num-snum[i]>j)l++ ;
           if(l<r)  f[i][j]=max(f[i][j] , q[l].mon+(q[l].num-j)*s[i]) ;
        }
     }
     int ans=-inf;
     for(i=0 ; i<=m ;i++)
     {
        if(f[n][i]>ans)ans=f[n][i] ;
     }
    
     printf("%d\n" , ans ) ;
  }
 return 0 ;   
}

 

http://acm.hdu.edu.cn/showproblem.php?pid=2993这个是一道很经典的斜率优化题，看了周源的论文，里面的解释很详细的。

# include <iostream>
#include <cstring>
#include <cstdio>
using namespace std ;
const int n_max=100002 ;
int q[n_max] , head , tail ;
double sum[n_max] ;
int n , len ;
double max(double x ,double y){return x>y ? x : y ;}
int get()
{
   int s=0 ;
   char ch=getchar() ;
   while(ch<'0' || ch>'9')ch=getchar() ;
   while(ch<='9' && ch>='0')
   {
      s=s*10+ch-'0' ;
      ch=getchar() ;
   }
   return s ;
}
int main()
{
  int i , j ;
  double ans ;
  while(~scanf("%d %d", &n ,&len))
  {
     sum[0]=0 ;
     for(i=1 ;i<=n ; i++)
     {
         int x ;
         x=get() ;
         sum[i]=sum[i-1]+x ;
     }
     head=tail =0 ;
     ans=0 ;

     for(i=len ; i<=n ;i++)
     {
        int now=i-len ;
        while(head<tail) //去掉上凸点
        {
           double y1= sum[q[tail]]-sum[q[tail-1]] , y2=sum[now]-sum[q[tail]] ;
           double x1=q[tail]-q[tail-1] , x2=now-q[tail] ;
           if(y1*x2>=y2*x1)tail-- ;
           else break ;
        }
        q[++tail]=now ;//压入队列的元素始终是符合j-i+1>=k的
        while(head<tail)//在去掉上凸点的条件下选最优的
        {
           double y1=sum[q[head]]-sum[i];
           double x1=q[head]-i;
           double y2=sum[q[head+1]]-sum[i];
           double x2=q[head+1]-i;
           if(y1*x2<=y2*x1)head++;
           else break;
        }
        ans=max(ans , (sum[i]-sum[q[head]])/(i-q[head])) ;
     }
     printf("%.2lf\n" , ans );
  }
  return 0 ;
}

 http://acm.hdu.edu.cn/showproblem.php?pid=3530再加上一题吧，这样就算完整了，做过的都在这里了，都已经有点模式化了。

# include <iostream>
using namespace std ;
const int n_max=100010 ;
int a[n_max] ;
int q1[n_max] , q2[n_max] , head1 , tail1 , head2 , tail2 ;

int main()
{
  int n , m , k , t , ans ;
  int i , j ;
  while(scanf("%d %d %d",&n , &m , &k )!=EOF)
  {
     for(i=0 ; i<n ; i++)  scanf("%d", &a[i]) ;
     head1=1 , tail1=0 , head2=1 , tail2=0 ;
     t=0 , ans=0 ;//这里的t一定要初始化，而且要根据自己定义的数组来初始化
     for(i=0 ; i<n ;i++)
     {
        while(head1<=tail1 && a[q1[tail1]]>=a[i])tail1-- ;//因为要求最长的，所以等号不能少
           q1[++tail1]=i ;
        while(head2<=tail2 && a[q2[tail2]]<=a[i])tail2-- ;
           q2[++tail2]=i ;
        while(a[q2[head2]]-a[q1[head1]]>k)
        {
           if(q1[head1]<q2[head2])  { t=q1[head1]+1 , head1++ ;}
           else {t=q2[head2]+1 , head2++ ;}
        }
        int sum=a[q2[head2]]-a[q1[head1]] ;
        if(sum>=m && sum<=k )
        {
           int pos=i-t+1 ;
           if(ans<pos) ans=pos ;
        }
     }
     printf("%d\n" , ans) ;
  }
  return 0 ;
}
