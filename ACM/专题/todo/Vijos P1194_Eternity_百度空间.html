P1194

Domino一个很经典的状态压缩和矩阵转移应用

搜索构造转移矩阵，a->b条件 （二进制） a为1的位b一定要为0，a为0的位若b为1连续出现的次数一定是偶数

本来过程名叫ML。MM。想想太邪恶了~~就给改掉了

Program P1194;
type
line=array [0..63] of longint;
matrix=array [0..63,0..63] of longint;
var
n,m,p:longint;
ans:line;
q:matrix;

Function Check(a,b:longint):boolean;
var
i:longint;
begin
for i:=0 to m-1 do
    if (a shr i) and 1=1 then
      begin
        if (b shr i) and 1=1 then exit(false);
      end else
        b:=b xor 1 shl i;
while b>0 do
    begin
      if b and 1=1 then
        begin
          if b and 3<>3 then exit(false);
          b:=b shr 1;
        end;
      b:=b shr 1;
    end;
exit(true);
end;

Procedure MakeMatrix;
var
i,j:longint;
begin
fillchar(q,sizeof(q),0);
for i:=0 to 1 shl m-1 do
    for j:=0 to 1 shl m-1 do
      if Check(i,j) then q[i,j]:=1;
end;

Function LM(a:line;b:matrix):line;
var
i,j:longint;
begin
fillchar(LM,sizeof(LM),0);
for i:=0 to 1 shl m-1 do
    for j:=0 to 1 shl m-1 do
      begin
        inc(LM[i],a[j]*b[j,i]);
        LM[i]:=LM[i] mod p;
      end;
end;

Function MM(a,b:matrix):matrix;
var
i,j,k:longint;
begin
fillchar(MM,sizeof(MM),0);
for i:=0 to 1 shl m-1 do
    for j:=0 to 1 shl m-1 do
      for k:=0 to 1 shl m-1 do
        begin
          inc(MM[i,j],a[i,k]*b[k,j]);
          MM[i,j]:=MM[i,j] mod p;
        end;
end;

begin
readln(n,m,p);
MakeMatrix;
fillchar(ans,sizeof(ans),0);
ans[0]:=1;
while n>0 do
    begin
      if n and 1=1 then ans:=LM(ans,q);
      n:=n shr 1;
      q:=MM(q,q);
    end;
writeln(ans[0]);
end.
